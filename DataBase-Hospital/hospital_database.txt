// -- Tabela de tipo_medico
table tipo_medico {
  id int [primary key]
  nome varchar
}

// -- Tabela de especialidade
table especialidade {
  id int [primary key]
  nome varchar
}

// -- Tabela de convenio
table convenio {
  id int [primary key]
  nome varchar
  cnpj int
  tempo_carencia int
}

// -- Tabela de medicos
table medicos {
  id int [primary key]
  nome varchar
  telefone int
  email varchar
  tipo_medico_id int [ref: > tipo_medico.id]
  especialidade_id int [ref: > especialidade.id]
}

// -- Tabela de paciente
table paciente {
  id int [primary key]
  nome varchar
  data_nascimento int
  telefone int
  email varchar
  cpf int
  rg int
  convenio_id int [ref: > convenio.id]
  medico_id int [ref: > medicos.id]
}

// -- Tabela de consulta
table consulta {
  id int [primary key]
  data_hora int
  medico_id int [ref: > medicos.id]
  paciente_id int [ref: > paciente.id]
  especialidade_id int [ref: > especialidade.id]
  valor_consulta int
  numero_carteira int
}

// -- Tabela de receita
table receita {
  id int [primary key]
  consulta_id int [ref: > consulta.id]
  medicamento varchar
  quantidade int
  intrucoes_uso varchar
}

// -- Tabela de quarto
table quarto {
  id int [primary key]
  numero int
  tipo_quarto_id int [ref: > tipo_quarto.id]
}

// -- Tabela de tipo_quarto
table tipo_quarto {
  id int [primary key]
  descricao varchar
  valor_diaria int
}

// -- Tabela de enfermeiro
table enfermeiro {
  id int [primary key]
  nome varchar
  cpf int
  cre varchar
}

// -- Tabela de internacao
table internacao {
  id int [primary key]
  data_entrada int
  data_prev_alta int
  data_alta int
  procedimento varchar
  medico_id int [ref: > medicos.id]
  paciente_id int [ref: > paciente.id]
  enfermeiro_id int [ref: > enfermeiro.id]
  quarto_id int [ref: > quarto.id]
}
